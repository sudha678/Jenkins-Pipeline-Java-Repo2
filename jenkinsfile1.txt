pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                echo 'Building java source code webhook..'
		sh '''javac jenkins.java
		java jenkins
		'''
		 //echo 'Fail!'; exit 1
            }
        }
        stage('Test') {
            steps {
                echo 'Testing java source code webhook'
            }
        }
        stage('package') {
            steps {
                echo 'Creating java package webhook'
            }
        }
	stage('deploy') {
		steps {
			echo 'Deploying java package webhook....'
		}
            }
        }
    
    post {
        always {
            echo 'This code will always run webhook'
        }
        success {
            echo 'This code will only be executed on success webhook'
        }
        failure {
            echo 'This code will only be executed on failure webhook'
        }
        changed {
            echo 'This code will only be executed on change webhook.'
        }
    }
}
